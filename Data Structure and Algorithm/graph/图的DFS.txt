#include <iostream>
#include <vector>
#include <stack>
using namespace std;

// 用栈实现 DFS，图使用邻接表表示
void DFS(int start, const vector<vector<int>>& adj, vector<bool>& visited) {
    stack<int> s;
    s.push(start);

    while (!s.empty()) {
        int node = s.top();
        s.pop();

        if (!visited[node]) {
            visited[node] = true;
            cout << node << " ";  // 访问该节点

            // 将邻接点压栈（从后往前压入，可以控制遍历顺序）
            for (int i = adj[node].size() - 1; i >= 0; i--) {
                int neighbor = adj[node][i];
                if (!visited[neighbor]) {
                    s.push(neighbor);
                }
            }
        }
    }
}

int main() {
    int n = 4;  // 节点数量
    vector<vector<int>> adj(n);
    
    // 建立邻接表
    adj[0] = {1, 3};
    adj[1] = {2};
    adj[2] = {};
    adj[3] = {};

    vector<bool> visited(n, false);
    cout << "DFS traversal from node 0: ";
    DFS(0, adj, visited);  // 从节点0开始
    cout << endl;

    return 0;
}
