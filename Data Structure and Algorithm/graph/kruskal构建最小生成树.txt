#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 定义边结构
struct Edge {
    int u, v, weight;
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

// 并查集实现
class UnionFind {
    vector<int> parent;
public:
    UnionFind(int n) {
        parent.resize(n);
        for(int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);  // 路径压缩
        return parent[x];
    }
    void unite(int x, int y) {
        parent[find(x)] = find(y);
    }
    bool connected(int x, int y) {
        return find(x) == find(y);
    }
};

int main() {
    int n = 4;  // 节点数
    vector<Edge> edges = {
        {0, 1, 1},
        {1, 2, 4},
        {0, 2, 3},
        {2, 3, 2},
        {1, 3, 5}
    };

    sort(edges.begin(), edges.end());  // 按权重排序
    UnionFind uf(n);

    vector<Edge> mst;
    int totalCost = 0;

    for (auto& e : edges) {
        if (!uf.connected(e.u, e.v)) {
            uf.unite(e.u, e.v);
            mst.push_back(e);
            totalCost += e.weight;
        }
    }

    cout << "最小生成树边如下:\n";
    for (auto& e : mst) {
        cout << e.u << " - " << e.v << " : " << e.weight << endl;
    }
    cout << "总权重: " << totalCost << endl;

    return 0;
}
