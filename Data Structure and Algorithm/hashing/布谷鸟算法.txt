#include <iostream>
#include <vector>
#include <cmath>
#include <cstdlib>
#include <ctime>

using namespace std;

const int TABLE_SIZE = 11;         // 初始哈希表大小
const int MAX_KICKOUTS = 20;       // 防止无限循环的最大踢出次数

class CuckooHash {
private:
    vector<int> table1;
    vector<int> table2;
    int size;

    // 哈希函数1：对 table1
    int hash1(int key) {
        return key % size;
    }

    // 哈希函数2：对 table2
    int hash2(int key) {
        return (key / size) % size;
    }

public:
    CuckooHash(int sz = TABLE_SIZE) {
        size = sz;
        table1.assign(size, -1);
        table2.assign(size, -1);
    }

    bool search(int key) {
        return (table1[hash1(key)] == key || table2[hash2(key)] == key);
    }

    void print() {
        cout << "Table 1: ";
        for (int i = 0; i < size; i++) cout << table1[i] << " ";
        cout << "\nTable 2: ";
        for (int i = 0; i < size; i++) cout << table2[i] << " ";
        cout << "\n";
    }

    bool insert(int key) {
        if (search(key)) return true;

        int currentKey = key;
        int tableId = 1;
        for (int attempt = 0; attempt < MAX_KICKOUTS; attempt++) {
            if (tableId == 1) {
                int pos = hash1(currentKey);
                if (table1[pos] == -1) {
                    table1[pos] = currentKey;
                    return true;
                } else {
                    swap(currentKey, table1[pos]);
                    tableId = 2;
                }
            } else {
                int pos = hash2(currentKey);
                if (table2[pos] == -1) {
                    table2[pos] = currentKey;
                    return true;
                } else {
                    swap(currentKey, table2[pos]);
                    tableId = 1;
                }
            }
        }

        // 重哈希（未实现，简化演示）
        cout << "Insertion failed for " << key << ". Consider rehashing!\n";
        return false;
    }
};
